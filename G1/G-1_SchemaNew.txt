CREATE TABLE  "NEWSFEED" 
   (	"NEWS_ID" VARCHAR2(20) NOT NULL ENABLE, 
	"NEWS_TOPIC" VARCHAR2(400) NOT NULL ENABLE, 
	"NEWS_DATE" VARCHAR2(20) NOT NULL ENABLE, 
	"NEWS_CAT" VARCHAR2(10) NOT NULL ENABLE, 
	"NEWS_LINK" CLOB NOT NULL ENABLE
   );
CREATE TABLE  "LOGIN" 
   (	"USER_ID" VARCHAR2(40) NOT NULL ENABLE, 
	"USER_PASS" VARCHAR2(40) NOT NULL ENABLE, 
	"USER_CAT" VARCHAR2(10) NOT NULL ENABLE, 
	"USER_NAME" VARCHAR2(40) NOT NULL ENABLE
   );
CREATE TABLE  "NOTIFICATION"

	"NOTIFY_TYPE" VARCHAR2(40) NOT NULL ENABLE, 
	"NOTIFY_DATE" VARCHAR2(20) NOT NULL ENABLE
   );
